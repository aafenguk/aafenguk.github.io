如何在Express.js中使用Pasport.js进行OAuth(Google,Facebook等)登录

现在在Web应用程序中使用OAuth登录非常普遍。好处可以说是太多了。对于用户来说，避免再记一个密码。 对我服务方来说，避免保存用户信息，增加安全性，同时避免了很多麻烦，比如数据隐私。

技术栈：

  • Node.js
  • Express
  • MongoDB
  • Passport.js

配置基本Express应用

基本Express APP

首先创建基本的Express应用程序。

mkdir oauth-demo
cd oauth-demo
npm init

安装依赖库：

npm install -g nodemon
npm install express ejs mongoose passport passport-google-oauth20 cookie-session

创建相关目录

mkdir config
mkdir models
mkdir routes
mkdir views

添加基本文件

添加app.js:

const express = require('express');
const routes = require('./routes/index');

const app = express();

// 使用EJS模板
app.set('view engine', 'ejs');

// 定义路由
app.use('/', routes);

app.listen(3000, '0.0.0.0', () => {
    console.log('Listening on port 3000.');
});

module.exports = app;

添加文件routes/index.js:

const express = require('express');

const router = express.Router();

router.get('/', (req, res) => {
    res.render('index', { user: req.user });
});

module.exports = router;

添加文件：views/index.ejs:

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<body>
    Homepage
</body>
</html>

启动app：

nodemon app

就可以在3000端口访问应用了。

实现Google OAuth登录

添加routes/oauthRoutes.js

const router = require('express').Router();
const passport = require('passport');

// 注销
router.get('/logout', (req, res) => {
    req.logout();
    res.redirect('/');
});

// 通过Google登录
router.get('/google', passport.authenticate('google', {
    scope:['profile']
}));

// Google登陆后的回调
router.get('/google/redirect', passport.authenticate('google'), (req, res)=>{
    // passport会自动将user信息添加到req中
    // 登录完后自动跳转到/profile
    res.redirect('/profile');
})

module.exports = router;

注册oauth路由

在app.js中，

const oauthRoutes = require('./routes/oauthRoutes');
app.use('/auth', oauthRoutes);

创建models/user-model.js

const mongoose = require('mongoose');
const Schema = mongoose.Schema;

const userSchema = new Schema({
    username: String,
    provider_and_id: String,
    thumbnail: String
});

const User = mongoose.model('user', userSchema);

module.exports = User;

通过passport实现Google登录及用户创建/读取逻辑

const passport = require('passport');
const GoogleStrategy = require('passport-google-oauth20');
const credentials = require('./credentials');
const User = require('../models/user-model');

passport.serializeUser((user, done) => {
    done(null, user.id);
});

passport.deserializeUser((id, done) => {
    User.findById(id).then((user) => {
        done(null, user);
    });
});

passport.use(
    new GoogleStrategy({
        callbackURL: '/auth/google/redirect',
        clientID: credentials.google.clientID,
        clientSecret: credentials.google.clientSecret
    }, (accessToken, refreshToken, profile, done) => {
        // 如果用户已经存在
        User.findOne({provider_and_id: `google_${profile.id}`}).then((currentUser) => {
            if(currentUser){
                // already have this user
                console.log('user is: ', currentUser);
                // do something
                done(null, currentUser);
            } else {
                // if not, create user in our db
                console.log(profile);
                new User({
                    provider_and_id: `google_${profile.id}`,
                    username: profile.displayName,
                    thumbnail: profile._json.picture
                }).save().then((newUser) => {
                    console.log('created new user: ', newUser);
                    // do something
                    done(null, newUser);
                });
            }
        });
    })
);

在routes/oauthRoutes.js中：

// 通过Google登录
router.get('/google', passport.authenticate('google', {
    scope:['profile']
}));

// Google登陆后的回调，这里仍需要调用中间件passport.authenticate('google')，
// 用于和Google交换数据
router.get('/google/redirect', passport.authenticate('google'), (req, res)=>{
    // passport会自动将user信息添加到req中
    // 登录完后自动跳转到/profile

    res.redirect('/');
})

通过session/cookie保存登录信息，并保护特定页面

特别需要注意，添加cookie/session模块并在app.js中注册，否则passport无法保存登录状态！

const cookieSession = require('cookie-session');
// 使用Cookie
app.use(cookieSession({
    maxAge: 7 * 24 * 60 * 60 * 1000,  // Cookie保存一周
    keys: [credentials.session.cookieKey]
}));

在/profile路由中指定auth中间件来保护这个页面，文件：routes/index.js:

const authCheck = (req, res, next) => {
    // console.log(req.user);
    if(!req.user){
        res.redirect('/login');
    } else {
        next();
    }
};

router.get('/profile', authCheck, (req, res) => {
    res.render('profile', { user: req.user });
});

在profile.ejs中：

<% if (user) { %>
    <img src="<%= user.thumbnail %>" />
<% } %>

使用passport.js进行Facebook OAuth登录

创建facebook app并获取clientID & clientSecret

访问：https://developers.facebook.com/

需要注意的是，如果使用localhost:3000进行测试的话，需要创建一个Test App，而不是New App。在production环境下使用https时才能创建New App。

安装passport-facebook:

npm install passport-facebook

将facebook app的信息保存在config/credentials.js中：

facebook: {
  clientID: '',
  clientSecret: ''
},

在config/passport-init.js中，添加对FacebookStrtegy的初始化：

const FacebookStrategy = require('passport-facebook').Strategy;

passport.use(new FacebookStrategy({
    callbackURL: '/auth/facebook/redirect',
    clientID: credentials.facebook.clientID,
    clientSecret: credentials.facebook.clientSecret
},
function(accessToken, refreshToken, profile, done) {
    // 如果用户已经存在
    User.findOne({provider_and_id: `facebook_${profile.id}`}).then((currentUser) => {
        if(currentUser){
            // already have this user
            console.log('user is: ', currentUser);
            // do something
            done(null, currentUser);
        } else {
            // if not, create user in our db
            console.log(profile);
            new User({
                provider_and_id: `facebook_${profile.id}`,
                username: profile.displayName,
                thumbnail: profile.profileUrl
            }).save().then((newUser) => {
                console.log('created new user: ', newUser);
                // do something
                done(null, newUser);
            });
        }
    });
}
));
